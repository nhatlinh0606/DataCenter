*** add  sp_addlinkedserver (Transact-SQL)

EXEC sp_addlinkedserver   
   @server = N'e3e62ed12a1f',  
   @srvproduct=N'',  
   @provider=N'SQLNCLI',  
   @datasrc=N'e3e62ed12a1f\ACCTG';
